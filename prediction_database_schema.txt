-- Prediction Markets Database Schema
-- Run this first to create all required tables

-- 1. Markets table - stores prediction market information
CREATE TABLE IF NOT EXISTS prediction_markets (
    id SERIAL PRIMARY KEY,
    title VARCHAR(500) NOT NULL,
    description TEXT NOT NULL,
    category VARCHAR(100),
    creator_address VARCHAR(100) NOT NULL,
    resolver_address VARCHAR(100),
    start_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    end_date TIMESTAMP NOT NULL,
    resolution_date TIMESTAMP,
    status VARCHAR(50) DEFAULT 'upcoming' CHECK (status IN ('upcoming', 'active', 'closed', 'resolved', 'cancelled')),
    market_type VARCHAR(50) DEFAULT 'binary' CHECK (market_type IN ('binary', 'multiple', 'scalar')),
    min_bet DECIMAL(20,8) DEFAULT 1,
    max_bet DECIMAL(20,8) DEFAULT 1000,
    total_volume DECIMAL(20,8) DEFAULT 0,
    platform_fee_percentage DECIMAL(5,2) DEFAULT 2.0,
    oracle_fee_percentage DECIMAL(5,2) DEFAULT 1.0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. Outcomes table - stores possible outcomes for each market
CREATE TABLE IF NOT EXISTS prediction_outcomes (
    id SERIAL PRIMARY KEY,
    market_id INTEGER NOT NULL REFERENCES prediction_markets(id) ON DELETE CASCADE,
    outcome_name VARCHAR(200) NOT NULL,
    outcome_description TEXT,
    current_price DECIMAL(10,8) DEFAULT 0.5,
    total_shares DECIMAL(20,8) DEFAULT 0,
    is_winning BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(market_id, outcome_name)
);

-- 3. Positions table - tracks user positions in markets
CREATE TABLE IF NOT EXISTS prediction_positions (
    id SERIAL PRIMARY KEY,
    market_id INTEGER NOT NULL REFERENCES prediction_markets(id) ON DELETE CASCADE,
    user_address VARCHAR(100) NOT NULL,
    outcome_id INTEGER NOT NULL REFERENCES prediction_outcomes(id) ON DELETE CASCADE,
    shares_owned DECIMAL(20,8) DEFAULT 0,
    avg_buy_price DECIMAL(10,8) DEFAULT 0,
    total_invested DECIMAL(20,8) DEFAULT 0,
    realized_profit DECIMAL(20,8) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(market_id, user_address, outcome_id)
);

-- 4. Trades table - records all buy/sell transactions
CREATE TABLE IF NOT EXISTS prediction_trades (
    id SERIAL PRIMARY KEY,
    market_id INTEGER NOT NULL REFERENCES prediction_markets(id) ON DELETE CASCADE,
    user_address VARCHAR(100) NOT NULL,
    outcome_id INTEGER NOT NULL REFERENCES prediction_outcomes(id) ON DELETE CASCADE,
    trade_type VARCHAR(10) NOT NULL CHECK (trade_type IN ('buy', 'sell')),
    shares DECIMAL(20,8) NOT NULL,
    price DECIMAL(10,8) NOT NULL,
    total_amount DECIMAL(20,8) NOT NULL,
    platform_fee DECIMAL(20,8) DEFAULT 0,
    oracle_fee DECIMAL(20,8) DEFAULT 0,
    tx_id VARCHAR(200),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 5. Resolutions table - audit trail for market resolutions
CREATE TABLE IF NOT EXISTS prediction_resolutions (
    id SERIAL PRIMARY KEY,
    market_id INTEGER NOT NULL REFERENCES prediction_markets(id) ON DELETE CASCADE,
    resolver_address VARCHAR(100) NOT NULL,
    winning_outcome_id INTEGER REFERENCES prediction_outcomes(id),
    resolution_type VARCHAR(50) CHECK (resolution_type IN ('normal', 'invalid', 'tie')),
    notes TEXT,
    tx_id VARCHAR(200),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create indexes for better query performance
CREATE INDEX IF NOT EXISTS idx_markets_status ON prediction_markets(status);
CREATE INDEX IF NOT EXISTS idx_markets_category ON prediction_markets(category);
CREATE INDEX IF NOT EXISTS idx_markets_end_date ON prediction_markets(end_date);
CREATE INDEX IF NOT EXISTS idx_markets_creator ON prediction_markets(creator_address);

CREATE INDEX IF NOT EXISTS idx_outcomes_market ON prediction_outcomes(market_id);
CREATE INDEX IF NOT EXISTS idx_outcomes_winning ON prediction_outcomes(is_winning);

CREATE INDEX IF NOT EXISTS idx_positions_market ON prediction_positions(market_id);
CREATE INDEX IF NOT EXISTS idx_positions_user ON prediction_positions(user_address);
CREATE INDEX IF NOT EXISTS idx_positions_outcome ON prediction_positions(outcome_id);

CREATE INDEX IF NOT EXISTS idx_trades_market ON prediction_trades(market_id);
CREATE INDEX IF NOT EXISTS idx_trades_user ON prediction_trades(user_address);
CREATE INDEX IF NOT EXISTS idx_trades_outcome ON prediction_trades(outcome_id);
CREATE INDEX IF NOT EXISTS idx_trades_created ON prediction_trades(created_at);

CREATE INDEX IF NOT EXISTS idx_resolutions_market ON prediction_resolutions(market_id);

-- Function to automatically update updated_at timestamp
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ language 'plpgsql';

-- Create triggers for auto-updating updated_at
CREATE TRIGGER update_prediction_markets_updated_at BEFORE UPDATE ON prediction_markets
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_prediction_outcomes_updated_at BEFORE UPDATE ON prediction_outcomes
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_prediction_positions_updated_at BEFORE UPDATE ON prediction_positions
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
